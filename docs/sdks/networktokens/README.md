## list

List all network tokens stored for a payment method.

### Example Usage

```php
declare(strict_types=1);

require 'vendor/autoload.php';

use Gr4vy;

$sdk = Gr4vy\SDK::builder()
    ->setSecurity(
        '<YOUR_BEARER_TOKEN_HERE>'
    )
    ->setMerchantAccountId('default')
    ->build();



$response = $sdk->paymentMethods->networkTokens->list(
    paymentMethodId: 'ef9496d8-53a5-4aad-8ca2-00eb68334389',
    merchantAccountId: 'default'

);

if ($response->collectionNoCursorNetworkToken !== null) {
    // handle response
}
```

## create

Provision a network token for a payment method.

### Example Usage

```php
declare(strict_types=1);

require 'vendor/autoload.php';

use Gr4vy;

$sdk = Gr4vy\SDK::builder()
    ->setSecurity(
        '<YOUR_BEARER_TOKEN_HERE>'
    )
    ->setMerchantAccountId('default')
    ->build();

$networkTokenCreate = new Gr4vy\NetworkTokenCreate(
    securityCode: '123',
    merchantInitiated: false,
    isSubsequentPayment: false,
);

$response = $sdk->paymentMethods->networkTokens->create(
    paymentMethodId: 'ef9496d8-53a5-4aad-8ca2-00eb68334389',
    networkTokenCreate: $networkTokenCreate,
    merchantAccountId: 'default'

);

if ($response->networkToken !== null) {
    // handle response
}
```
